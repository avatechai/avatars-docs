```jsx
const blendshapesService = new BlendshapesService(
  [new MediapipeSourceHandler(), new VolumeSourceHandler()],
  [
    new BlendshapeHandler(
      (
        inputBlendshapes: Blendshapes,
        previousBlendshapes: Blendshapes,
        currentFrame: number
      ) => {
        const {
          Volume,
          Jaw_L,
          Jaw_R,
          Mouth_Open,
          Mouth_L,
          Mouth_R,
          Lips_Down,
          Lips_Up,
          Brow_Lower_L,
          Brow_Raise_L,
          Brow_Lower_R,
          Brow_Raise_R,
          Eye_Close_L,
          Eye_Open_L,
          Eye_Close_R,
          Eye_Open_R,
          Kiss,
          Smile_L,
          Smile_R,
          Lips_Corner_Lower_L,
          Lips_Corner_Lower_R,
          Upper_Lip_Raise_L,
          Upper_Lip_Raise_R,
          Eye_Move_Up_L,
          Eye_Move_Down_L,
          Eye_Move_In_L,
          Eye_Move_Out_L,
          Eye_Move_Up_R,
          Eye_Move_Down_R,
          Eye_Move_In_R,
          Eye_Move_Out_R,
        } = inputBlendshapes
        return {
          ParamMouthFunnel: BezierEasing(
            0.0968149784239621,
            0.6623859252929687,
            0.3820135343806408,
            0.33038592529296873
          )(Kiss * 10000),
          ParamMouthOpenY: BezierEasing(
            0.4641146332778655,
            0.16340881347656255,
            0.15725542750169225,
            0.8954085693359375
          )(Mouth_Open * 4),
          ParamMouthPressLipOpen: BezierEasing(
            0.27024692576714804,
            0.8553101806640625,
            0.8534162500705099,
            0.21955749511718747
          )(Math.max(Lips_Corner_Lower_R + Lips_Down) * 100),
          ParamMouthPuckerWiden: BezierEasing(
            0.06295346697314982,
            0.5040351562500001,
            0.19291736589011732,
            0.5960351562499999
          )(Smile_L + Smile_R),
          ParamMouthShrug: BezierEasing(
            0.21490742927854242,
            0.76899365234375,
            0.7989535004653655,
            0.319709716796875
          )((Mouth_L + Mouth_R) * 100),
          ParamBrowLY: BezierEasing(
            0.27692906348713897,
            0.03488790893554683,
            0.6812611934510379,
            0.3508879089355469
          )(Volume),
          ParamBrowRY: BezierEasing(
            0.27692906348713897,
            0.03488790893554683,
            0.6812611934510379,
            0.3508879089355469
          )(Volume),
          ParamEyeLOpen: BezierEasing(
            0.11157821490015794,
            0.6911943359375,
            0.5,
            1
          )(
            currentFrame % (10 + 100) >= 10
              ? 1
              : (currentFrame % (10 + 100)) / 10
          ),
          ParamEyeROpen: BezierEasing(
            0.11157821490015794,
            0.6911943359375,
            0.5,
            1
          )(
            currentFrame % (10 + 100) >= 10
              ? 1
              : (currentFrame % (10 + 100)) / 10
          ),
        }
      }
    ),
  ]
)
```
